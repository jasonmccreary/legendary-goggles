## Aliases

alias.unstage reset HEAD --
alias.logo log --oneline
alias.append commit --amend --no-edit
alias.lola log --graph --oneline --decorate --all
alias.wip commit -m "WIP"
alias.cpl log --oneline -3 --name-status
